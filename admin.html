<script>

 // ============================================================

 // CONFIGURAÇÃO — mesma URL do Apps Script

 // ============================================================

 const APPS_SCRIPT_URL = 'https://script.google.com/a/macros/gestaoonsaude.com/s/AKfycby-Y9yiYAnZyd3J6qNkWAZRwL1UzRUi8OZOXbXdvUkDq2isDLBf-b1eQVfxHgUnHHup/exec';

 // ============================================================

 

 let dados = [];

 let filtrados = [];

 let idxAtual = null;

 

 async function carregarDados() {

   const btn = document.getElementById('reload-btn');

   btn.disabled = true;

   btn.innerHTML = '<span class="spinner"></span> Carregando...';

   setTableLoading();

   try {

     const resp = await fetch(APPS_SCRIPT_URL + '?action=listar');

     const json = await resp.json();

     dados = json.dados || [];

     filtrar();

   } catch(e) {

     document.getElementById('tbody').innerHTML = '<tr class="empty-row"><td colspan="7">Erro ao carregar dados. Verifique a conexão com o servidor.</td></tr>';

   }

   btn.disabled = false;

   btn.innerHTML = '<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2.5" stroke-linecap="round" stroke-linejoin="round"><polyline points="23 4 23 10 17 10"/><path d="M20.49 15a9 9 0 1 1-2.12-9.36L23 10"/></svg> Atualizar';

 }

 

 function setTableLoading() {

   document.getElementById('tbody').innerHTML = '<tr class="empty-row"><td colspan="7">Carregando manifestações...</td></tr>';

 }

 

 function filtrar() {

   const fu = document.getElementById('f-unidade').value;

   const ft = document.getElementById('f-tipo').value;

   const fs = document.getElementById('f-status').value;

   filtrados = dados.filter(r =>

     (!fu || r.unidade === fu) &&

     (!ft || r.tipo === ft) &&

     (!fs || r.status === fs)

   );

   renderTabela();

   atualizarMetricas();

 }

 

 function statusBadge(s) {

   const map = { 'Novo': 'badge-novo', 'Em análise': 'badge-analise', 'Respondido': 'badge-respondido' };

   return `<span class="badge ${map[s] || 'badge-novo'}">${s}</span>`;

 }

 

 function anexoCell(row) {

   if (!row.anexos || row.anexos.length === 0) return '<span class="no-attach">—</span>';

   return `<div class="dl-multi">${row.anexos.map(a =>

     `<a class="dl-btn" href="${a.url}" target="_blank" title="${a.nome}">

       <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"/><polyline points="7 10 12 15 17 10"/><line x1="12" y1="15" x2="12" y2="3"/></svg>

       ${a.nome.length > 14 ? a.nome.substring(0,14)+'…' : a.nome}

     </a>`).join('')}</div>`;

 }

 

 function renderTabela() {

   const tbody = document.getElementById('tbody');

   if (!filtrados.length) {

     tbody.innerHTML = '<tr class="empty-row"><td colspan="7">Nenhuma manifestação encontrada.</td></tr>';

     return;

   }

   tbody.innerHTML = filtrados.map((r, i) => `

     <tr onclick="abrirModal(${dados.indexOf(r)})">

       <td><strong>${r.protocolo}</strong></td>

       <td>${r.tipo}</td>

       <td style="font-size:12px; color:var(--gray-600);">${r.unidade}</td>

       <td>${r.nome}</td>

       <td>${statusBadge(r.status)}</td>

       <td style="font-size:12px; color:var(--gray-600); white-space:nowrap;">${r.data}</td>

       <td onclick="event.stopPropagation()">${anexoCell(r)}</td>

     </tr>`).join('');

 }

 

 function atualizarMetricas() {

   document.getElementById('m-total').textContent = dados.length;

   document.getElementById('m-novo').textContent = dados.filter(r => r.status === 'Novo').length;

   document.getElementById('m-analise').textContent = dados.filter(r => r.status === 'Em análise').length;

   document.getElementById('m-resp').textContent = dados.filter(r => r.status === 'Respondido').length;

 }

 

 function abrirModal(idx) {

   idxAtual = idx;

   const r = dados[idx];

   document.getElementById('m-protocolo').textContent = r.protocolo;

   document.getElementById('m-subtitulo').textContent = r.data + ' — ' + r.tipo;

   document.getElementById('m-unidade').textContent = r.unidade;

   document.getElementById('m-tipo').textContent = r.tipo;

   document.getElementById('m-nome').textContent = r.nome;

   document.getElementById('m-contato').textContent = r.contato;

   document.getElementById('m-data').textContent = r.data;

   document.getElementById('m-desc').textContent = r.descricao;

   document.getElementById('m-status-sel').value = r.status;

   const anexosEl = document.getElementById('m-anexos');

   if (!r.anexos || r.anexos.length === 0) {

     anexosEl.innerHTML = '<p style="font-size:13px; color:var(--gray-400);">Nenhum anexo enviado.</p>';

   } else {

     anexosEl.innerHTML = `<div class="dl-multi">${r.anexos.map(a =>

       `<a class="dl-btn" href="${a.url}" target="_blank">

         <svg width="13" height="13" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4"/><polyline points="7 10 12 15 17 10"/><line x1="12" y1="15" x2="12" y2="3"/></svg>

         ${a.nome}

       </a>`).join('')}</div>`;

   }

   document.getElementById('modal').classList.add('open');

 }

 

 function fecharModal() { document.getElementById('modal').classList.remove('open'); }

 

 async function salvarStatus() {

   if (idxAtual === null) return;

   const novoStatus = document.getElementById('m-status-sel').value;

   const protocolo = dados[idxAtual].protocolo;

   try {

     await fetch(APPS_SCRIPT_URL, {

       method: 'POST',

       body: JSON.stringify({ action: 'atualizarStatus', protocolo, status: novoStatus })

     });

     dados[idxAtual].status = novoStatus;

     filtrar();

     fecharModal();

   } catch(e) {

     alert('Erro ao salvar status. Tente novamente.');

   }

 }

 

 // Fecha modal ao clicar fora

 document.getElementById('modal').addEventListener('click', function(e) {

   if (e.target === this) fecharModal();

 });

 

 // Carrega ao abrir

 carregarDados();

</script>
