<script>

 // ============================================================

 // CONFIGURAÇÃO — substitua pela URL do seu Google Apps Script

 // ============================================================

 const APPS_SCRIPT_URL = 'https://script.google.com/a/macros/gestaoonsaude.com/s/AKfycby-Y9yiYAnZyd3J6qNkWAZRwL1UzRUi8OZOXbXdvUkDq2isDLBf-b1eQVfxHgUnHHup/exec';

 // ============================================================

 

 let anonimo = false;

 let arquivos = [];

 

 function toggleAnon() {

   anonimo = !anonimo;

   document.getElementById('switch').classList.toggle('on', anonimo);

   document.getElementById('id-fields').style.display = anonimo ? 'none' : 'block';

 }

 

 function handleFiles(files) {

   for (const f of files) {

     if (f.size > 10 * 1024 * 1024) { showError('O arquivo "' + f.name + '" excede 10MB.'); continue; }

     if (!arquivos.find(a => a.name === f.name)) arquivos.push(f);

   }

   renderFiles();

   document.getElementById('upload-area').classList.toggle('active', arquivos.length > 0);

 }

 

 function renderFiles() {

   document.getElementById('file-list').innerHTML = arquivos.map((f, i) => `

     <div class="file-item">

       <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="#1a237e" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><path d="M14 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V8z"/><polyline points="14 2 14 8 20 8"/></svg>

       <span>${f.name} <span style="color:var(--gray-600)">(${(f.size/1024).toFixed(0)} KB)</span></span>

       <button onclick="removeFile(${i})" title="Remover">✕</button>

     </div>`).join('');

 }

 

 function removeFile(i) {

   arquivos.splice(i, 1);

   renderFiles();

   document.getElementById('upload-area').classList.toggle('active', arquivos.length > 0);

 }

 

 function showError(msg) {

   const el = document.getElementById('error-msg');

   el.textContent = msg;

   el.style.display = 'block';

   el.scrollIntoView({ behavior: 'smooth', block: 'center' });

 }

 

 function clearError() {

   document.getElementById('error-msg').style.display = 'none';

 }

 

 async function toBase64(file) {

   return new Promise((res, rej) => {

     const r = new FileReader();

     r.onload = () => res(r.result.split(',')[1]);

     r.onerror = rej;

     r.readAsDataURL(file);

   });

 }

 

 async function enviar() {

   clearError();

   const unidade = document.getElementById('unidade').value;

   const tipo = document.getElementById('tipo').value;

   const descricao = document.getElementById('descricao').value.trim();

 

   if (!unidade) { showError('Selecione a unidade onde o fato aconteceu.'); return; }

   if (!tipo) { showError('Selecione o tipo de relato.'); return; }

   if (!descricao) { showError('Preencha a descrição detalhada.'); return; }

 

   const btn = document.getElementById('submit-btn');

   btn.disabled = true;

   btn.innerHTML = '<span class="spinner"></span>Enviando...';

 

   // Converte arquivos para base64

   const anexos = [];

   for (const f of arquivos) {

     const b64 = await toBase64(f);

     anexos.push({ nome: f.name, tipo: f.type, dados: b64 });

   }

 

   const payload = {

     unidade,

     tipo,

     descricao,

     anonimo,

     nome: anonimo ? 'Anônimo' : (document.getElementById('nome').value.trim() || 'Não informado'),

     contato: anonimo ? '-' : (document.getElementById('contato').value.trim() || '-'),

     anexos

   };

 

   try {

     const resp = await fetch(APPS_SCRIPT_URL, {

       method: 'POST',

       body: JSON.stringify(payload)

     });

     const data = await resp.json();

     if (data.status === 'ok') {

       document.getElementById('protocolo-display').textContent = data.protocolo;

       document.getElementById('form-wrap').style.display = 'none';

       document.getElementById('success-wrap').style.display = 'block';

     } else {

       showError('Erro ao enviar: ' + (data.message || 'Tente novamente.'));

       btn.disabled = false;

       btn.textContent = 'Enviar manifestação';

     }

   } catch (e) {

     showError('Não foi possível conectar ao servidor. Verifique sua conexão e tente novamente.');

     btn.disabled = false;

     btn.textContent = 'Enviar manifestação';

   }

 }

 

 function resetForm() {

   anonimo = false;

   arquivos = [];

   ['unidade','tipo','descricao','nome','contato'].forEach(id => {

     const el = document.getElementById(id);

     if (el) el.value = '';

   });

   document.getElementById('file-list').innerHTML = '';

   document.getElementById('upload-area').classList.remove('active');

   document.getElementById('switch').classList.remove('on');

   document.getElementById('id-fields').style.display = 'block';

   document.getElementById('form-wrap').style.display = 'block';

   document.getElementById('success-wrap').style.display = 'none';

   document.getElementById('submit-btn').disabled = false;

   document.getElementById('submit-btn').textContent = 'Enviar manifestação';

   clearError();

 }
